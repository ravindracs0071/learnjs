        Interface:
        to enforce a class to follow the specifications in a particular contract.
        interface ILogger {
            log(arg: any) : void;
        }
        
        class Logger implements ILogger {
            log(arg: any): void {
                console.log(arg);
            }
        }
        
        let logger = new Logger();
        logger.log("Logged data");

        Namespaces :
        known as internal modules, used to encapsulate features and objects that share a certain relationship.
        organize your code, cleaner way
        to declare namespace in TypeScript use namespace and export keywords.
        
        namespace Geometry {
            interface IVector {
                getPoint(x: number, y: number) : void;
            }
            export interface IVector2D {
             getPoint2D(x: number, y: number) : void;
            }
         
            export class Vector implements IVector, IVector2D {
                getPoint2D(x: number, y: number): void {
                    console.log("Method : getPoint2D");
                }
                
                getPoint(x: number, y: number): void {
                    console.log("Method : getPoint");
                }
            }
         }
         
         
         var inst: Geometry.IVector2D = new Geometry.Vector();
         inst.getPoint2D(1, 2);
         
         
         Rest syntax/parameter
         function with rest parameter -- 
         function add(...nums: number[]) {
            var total : number = 0;
            for(var i = 0; i < nums.length; i++) { 
                total += nums[i];
            }
            return total;
        }
        
        console.log(add(1, 2));
        console.log(add(1, 2, 3));
